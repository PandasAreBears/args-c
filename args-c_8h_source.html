276__maybe_unused static struct ac_status ac_command_parse(int const argc,
277 char const *const *const argv,
278 struct ac_command_spec const *const command,
279 struct ac_command *const args) {
280#define AC_STATUS(...) (struct ac_status){.single = command, ##__VA_ARGS__};
281
282 if(argv == NULL || command == NULL || args == NULL) {
283 return AC_STATUS(.code = AC_ERROR_INVALID_PARAMETER);
284 }
285
286 if(argc > MAX_NUM_ARGS) {
287 return AC_STATUS(.code = AC_ERROR_ARGUMENT_MAX_EXCEEDED, .context = (void *) (size_t) argc);
288 }
289
290 bzero(args, sizeof(*args));
291
292 // We're going to continually reference the length of these strings, so
293 // pre-populate an array for fast lookup.
294 size_t strlens[MAX_NUM_ARGS] = {0};
295 for(size_t i = 0; i < argc; i++) {
296 strlens[i] = strnlen(argv[i], MAX_STRING_LEN);
297 }
298
299 enum tag {
300 TAG_ARGUMENT,
301 TAG_SHORT_OPTION,
302 TAG_LONG_OPTION,
303 TAG_OPTION_VALUE,
304 };
305 enum tag tags[MAX_NUM_ARGS];
306 size_t n_arguments = 0;
307 size_t n_options = 0;
308 bool arguments_complete = false;
309 for(size_t i = 0; i < argc; i++) {
310 // need to check for only alpha characters because e.g. '-1' is a valid
311 // value.
312 if(strlens[i] >= 3 && argv[i][0] == '-' && argv[i][1] == '-' &&
313 _ac_string_is_alpha(&argv[i][2], strlens[i])) {
314 tags[i] = TAG_LONG_OPTION;
315 n_options++;
316 arguments_complete = true;
317 continue;
318 }
319
320 if(strlens[i] == 2 && argv[i][0] == '-' && _ac_char_is_alpha(argv[i][1])) {
321 tags[i] = TAG_SHORT_OPTION;
322 n_options++;
323 arguments_complete = true;
324 continue;
325 }
326
327 // Anything that's not a option name is implicitly an argument or value.
328 if(!arguments_complete) {
329 n_arguments++;
330 tags[i] = TAG_ARGUMENT;
331 } else {
332 tags[i] = TAG_OPTION_VALUE;
333 }
334 }
335
336 if(n_arguments > command->n_arguments) {
337 return AC_STATUS(.code = AC_ERROR_ARGUMENT_EXCEEDED_SPEC, .context = (void *) n_arguments);
338 }
339 if(n_arguments < command->n_arguments) {
340 return AC_STATUS(.code = AC_ERROR_ARGUMENT_EXPECTED_IN_SPEC,
341 .context = (void *) n_arguments);
342 }
343 if(n_options > MAX_NUM_OPTIONS) {
344 return AC_STATUS(.code = AC_ERROR_OPTION_TOO_MANY, .context = (void *) n_options);
345 }
346
347 struct ac_argument *const arguments =
348 n_arguments > 0 ? (struct ac_argument *) calloc(n_arguments, sizeof(*arguments)) : NULL;
349 if(n_arguments != 0 && arguments == NULL) {
350 return AC_STATUS(.code = AC_ERROR_MEMORY_ALLOC_FAILED);
351 }
352
353 // Arguments are assigned in the order that they appear in the command.
354 for(size_t i = 0; i < n_arguments; i++) {
355 arguments[i].value = strdup(argv[i]);
356 arguments[i].argument = &command->arguments[i];
357 }
358
359 struct ac_option *options =
360 n_options > 0 ? (struct ac_option *) calloc(n_options, sizeof(*options)) : NULL;
361 if(n_options != 0 && options == NULL) {
362 free(arguments);
363 return AC_STATUS(.code = AC_ERROR_MEMORY_ALLOC_FAILED);
364 }
365
366#define cleanup() \
367 free(arguments); \
368 free(options)
369
370 size_t options_idx = 0;
371 bool expecting_value = false;
372 size_t i = n_arguments;
373 for(; i < argc; i++) {
374 char const *const value = argv[i];
375 switch(tags[i]) {
376 case TAG_ARGUMENT: {
377 assert(false);
378 }
379 case TAG_LONG_OPTION:
380 case TAG_SHORT_OPTION: {
381 if(expecting_value) {
382 cleanup();
383 return AC_STATUS(.code = AC_ERROR_OPTION_VALUE_EXPECTED,
384 .context = (char *) argv[i - 1]);
385 }
386
387 struct ac_option *const option = &options[options_idx];
388
389 // Find the option that this maps to in the command spec.
390 for(size_t j = 0; j < command->n_options; j++) {
391 struct ac_option_spec const *const option_spec = &command->options[j];
392 if(tags[i] == TAG_SHORT_OPTION) {
393 if(option_spec->has_short_name && option_spec->short_name == value[1]) {
394 option->option = option_spec;
395 }
396 } else if(tags[i] == TAG_LONG_OPTION) {
397 if(0 == strncmp(option_spec->long_name, &value[2], strlens[i] - 2)) {
398 option->option = option_spec;
399 }
400 } else {
401 assert(false);
402 }
403 }
404
405 if(option->option == NULL) {
406 cleanup();
407 return AC_STATUS(.code = AC_ERROR_OPTION_NAME_NOT_IN_SPEC,
408 .context = (void *) value);
409 }
410
411 if(option->option->is_flag) {
412 options_idx++;
413 } else {
414 expecting_value = true;
415 }
416 break;
417 }
418 case TAG_OPTION_VALUE: {
419 if(!expecting_value) {
420 cleanup();
421 return AC_STATUS(.code = AC_ERROR_OPTION_NAME_EXPECTED,
422 .context = (char *) value);
423 }
424
425 struct ac_option *const option = &options[options_idx];
426 assert(options->option != NULL);
427
428 option->value = strdup(value);
429
430 expecting_value = false;
431 options_idx++;
432 break;
433 }
434 }
435 }
436
437 if(expecting_value) {
438 cleanup();
439 return AC_STATUS(.code = AC_ERROR_OPTION_VALUE_EXPECTED, .context = (void *) argv[i]);
440 }
441
442 // Make sure all the required options are present.
443 for(size_t i = 0; i < command->n_options; i++) {
444 struct ac_option_spec const *const option_spec = &command->options[i];
445 if(option_spec->required) {
446 bool found = false;
447 for(size_t j = 0; j < n_options; j++) {
448 if(options[j].option == option_spec) {
449 found = true;
450 break;
451 }
452 }
453
454 if(!found) {
455 cleanup();
456 return AC_STATUS(.code = AC_ERROR_OPTION_NAME_REQUIRED_IN_SPEC,
457 .context = option_spec->long_name);
458 }
459 }
460 }
461
462 args->n_arguments = n_arguments;
463 args->arguments = arguments;
464 args->n_options = n_options;
465 args->options = options;
466 args->command = command;
467
468 return AC_STATUS(.code = AC_ERROR_SUCCESS);
469#undef cleanup
470#undef AC_STATUS
471}

483__maybe_unused static struct ac_status
484ac_multi_command_parse(int const argc, char const *const *const argv,
485 struct ac_multi_command_spec const *const root,
486 struct ac_command *const args) {
487 if(argc == 0 || argv == NULL || root == NULL || args == NULL) {
488 return (struct ac_status) {.code = AC_ERROR_INVALID_PARAMETER, .multi = root};
489 }
490
491 if(argc > MAX_NUM_ARGS) {
492 return (struct ac_status) {.code = AC_ERROR_ARGUMENT_MAX_EXCEEDED,
493 .context = (void *) (size_t) argc,
494 .multi = root};
495 }
496
497 // Figure out how many of the arguments are multi-command names.
498 size_t n_command_names = 0;
499 for(size_t i = 0; i < argc; i++) {
500 size_t namelen = strnlen(argv[i], MAX_STRING_LEN);
501 if(namelen == 0 && n_command_names == 0) {
502 // Empty string is always an invalid start.
503 return (struct ac_status) {
504 .code = AC_ERROR_COMMAND_NAME_INVALID, .context = (void *) argv[i], .multi = root};
505 }
506 if(namelen == 0 || argv[i][0] == '-') {
507 break;
508 }
509 n_command_names++;
510 }
511
512 // Traverse the command tree to resolve the command.
513 struct ac_multi_command_spec const *curr_node = root;
514 struct ac_command_spec const *command = NULL;
515 size_t i = 0;
516 for(; (i < n_command_names) && (command == NULL); i++) {
517 char const *const curr_name = argv[i];
518 bool found = false;
519
520 for(size_t j = 0; j < curr_node->n_subcommands; j++) {
521 if(0 == strncmp(curr_node->subcommands[j].name, curr_name, MAX_STRING_LEN)) {
522 found = true;
523
524 if(curr_node->subcommands[j].type == COMMAND_SINGLE) {
525 command = curr_node->subcommands[j].single;
526 break;
527 }
528
529 // If this is the last iteration, then we expect the node to be single.
530 if(i + 1 == n_command_names) {
531 return (struct ac_status) {.code = AC_ERROR_COMMAND_NAME_REQUIRED,
532 .multi = curr_node,
533 .context = (char *) curr_name};
534 }
535
536 // Otherwise we've found a matching subcommand, progress to the next node.
537 curr_node = curr_node->subcommands[j].multi;
538 }
539 }
540
541 if(!found) {
542 return (struct ac_status) {.code = AC_ERROR_COMMAND_NAME_NOT_IN_SPEC,
543 .context = (void *) curr_name,
544 .multi = curr_node};
545 }
546 }
547
548 assert(command != NULL);
549
550 return ac_command_parse(argc - i, &argv[i], command, args);
551}

553static size_t _ac_strcpy_safe(char dst[], char const src[], size_t const offset,
554 size_t const dst_len) {
555 size_t const src_len = strnlen(src, MAX_STRING_LEN);
556
557 if(offset + src_len + 1 >= dst_len) {
558 return 0;
559 }
560
561 memcpy(&dst[offset], src, src_len);
562 dst[offset + src_len] = '\0';
563 return src_len;
564}

747__maybe_unused static struct ac_status
748ac_command_validate(struct ac_command_spec const *const command) {
749 if(command == NULL) {
750 return (struct ac_status) {.code = AC_ERROR_INVALID_PARAMETER};
751 }
752
753 for(size_t i = 0; i < command->n_arguments; i++) {
754 struct ac_argument_spec const *const arg = &command->arguments[i];
755 if(arg->name == NULL) {
756 return (struct ac_status) {.code = AC_ERROR_ARGUMENT_SPEC_NEEDS_NAME,
757 .context = (void *) i};
758 }
759 }
760
761 for(size_t i = 0; i < command->n_options; i++) {
762 struct ac_option_spec const *const option = &command->options[i];
763 if(option->long_name == NULL) {
764 return (struct ac_status) {.code = AC_ERROR_OPTION_SPEC_NEEDS_NAME,
765 .context = (void *) i};
766 }
767 if(!_ac_string_is_alpha(option->long_name, strnlen(option->long_name, MAX_STRING_LEN))) {
768 return (struct ac_status) {.code = AC_ERROR_OPTION_LONG_NAME_INVALID,
769 .context = (void *) i};
770 }
771 if(option->has_short_name && !_ac_char_is_alpha(option->short_name)) {
772 return (struct ac_status) {.code = AC_ERROR_OPTION_SHORT_NAME_INVALID,
773 .context = (void *) i};
774 }
775 if(option->is_flag && option->required) {
776 return (struct ac_status) {.code = AC_ERROR_OPTION_FLAG_AND_REQUIRED,
777 .context = (void *) i};
778 }
779 }
780
781 return (struct ac_status) {.code = AC_ERROR_SUCCESS};
782}

881 case AC_ERROR_OPTION_TOO_MANY:
882 include_help = true;
883 errorf("Too many options provided.\n");
884 case AC_ERROR_OPTION_SPEC_NEEDS_NAME:

886 case AC_ERROR_OPTION_LONG_NAME_INVALID:

888 case AC_ERROR_OPTION_SHORT_NAME_INVALID:
889 errorf("Programmer error: Option in spec has an invalid short name field.\n");
890 case AC_ERROR_OPTION_FLAG_AND_REQUIRED:
891 errorf("Programmer error: Option in spec has both required and is_flag.\n");
892 case AC_ERROR_COMMAND_NAME_NOT_IN_SPEC:
893 include_help = true;
894 errorf("The command '%s' is not defined.\n", (char *) result.context);
895 case AC_ERROR_COMMAND_NAME_REQUIRED:
896 include_help = true;
897 errorf("Another command name is expected after %s.\n", (char *) result.context);
898 case AC_ERROR_COMMAND_NAME_INVALID:
899 include_help = true;
900 errorf("The command name %s is invalid.\n", (char *) result.context);
901 case AC_ERROR_ARGUMENT_MAX_EXCEEDED:
902 include_help = true;
903 errorf("Exceeded the maximum allowed number of arguments.\n");
904 case AC_ERROR_ARGUMENT_EXCEEDED_SPEC:
905 include_help = true;
906 errorf("Too many arguments. Got %zu which is more than expected.\n",
907 (size_t) result.context);
908 case AC_ERROR_ARGUMENT_EXPECTED_IN_SPEC:
909 include_help = true;
910 errorf("Missing arguments.\n");
911 case AC_ERROR_ARGUMENT_SPEC_NEEDS_NAME:
912 errorf("Programmer error: Argument at index %zu needs a name.\n",
913 (size_t) result.context);
914 case AC_ERROR_MULTICOMMAND_NEEDS_NAME:
915 errorf("Programmer error: Multi-command at index %zu needs a name.\n",
916 (size_t) result.context);
917 }
918#undef errorf
919
920 if(!include_help) {
921 return error;
922 }
923
924 if(result.single == NULL && result.multi == NULL) {
925 return NULL;
926 }
927
928 char *help =
929 result.single ? ac_command_help(result.single, NULL) : ac_multi_command_help(result.multi, NULL);
930 if(help == NULL) {
931 return NULL;
932 }
933
934 size_t cursor = strnlen(help, HELP_BUFFER_SZ);
935 cursor += _ac_strcpy_safe(help, "\n", cursor, HELP_BUFFER_SZ);
936 (void) _ac_strcpy_safe(help, error, cursor, HELP_BUFFER_SZ);
937 free(error);
938
939 return help;
940}
